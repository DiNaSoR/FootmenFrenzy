"item_upgrade_weapon1"

	"AbilityTextureName"			"item_upgrade_weapon1"

	{

		{
			"ScriptFile"	"upgrades.lua"
			"Function"		"UpgradeFinished"
			"Name"			"upgrade_weapon"
			"Level"			"1"
		}
	}
