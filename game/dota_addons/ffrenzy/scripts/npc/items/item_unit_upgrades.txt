// 3 Upgrades are accessed through Items in the base inventory.
// 100 gold +50 gold cost each rank, 20 levels.

"item_upgrade_weapon1"
{
    "BaseClass"             		"item_datadriven"
	"AbilityTextureName"			"item_upgrade_weapon1"

	"AbilityBehavior"				"DOTA_ABILITY_BEHAVIOR_NO_TARGET | DOTA_ABILITY_BEHAVIOR_IMMEDIATE"

	"ItemDroppable" 				"0"
	"ItemPurchasable" 				"0"
	"ItemSellable"					"0"

	"AbilityGoldCost"				"100"

	"OnSpellStart"
	{
		// Add Upgrade and Swap the ability for the next upgrade
		"RunScript"
		{
			"ScriptFile"	"upgrades.lua"
			"Function"		"UpgradeFinished"
			"Name"			"upgrade_weapon"
			"Level"			"1"
		}
	}
}
